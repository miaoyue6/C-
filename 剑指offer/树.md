#### 4.重建二叉树

``` c++
class Solution {
public:
    TreeNode* reConstructBinaryTree(vector<int> pre,vector<int> vin) {
        return constructTree(pre,0,pre.size()-1,vin,0,vin.size()-1);
    }
    TreeNode* constructTree(vector<int> pre,int pStart,int pEnd,vector<int> vin,int vStart,int vEnd){
        if(pStart>pEnd)
            return NULL;
        TreeNode* cur=new TreeNode(pre[pStart]);
        int vIndex=vStart;
        for(;vIndex<vEnd;vIndex++){
            if(vin[vIndex]==pre[pStart])
                break;
        }
        int leftNum=vIndex-vStart;
        
        cur->left=constructTree(pre,pStart+1,pStart+leftNum,vin,vStart,vIndex-1);
        cur->right=constructTree(pre,pStart+leftNum+1,pEnd,vin,vIndex+1,vEnd);
        return cur;
        
    }
    
};
```

#### 17.树的子结构

``` c++
class Solution {
public:
    bool HasSubtree(TreeNode* pRoot1, TreeNode* pRoot2)
    {
        if(pRoot1==NULL||pRoot2==NULL)
            return false;
        bool flag=false;
        if(pRoot1->val==pRoot2->val)
            flag= dfs(pRoot1,pRoot2);
        if(flag)
            return true;
        return HasSubtree(pRoot1->left,pRoot2)|| HasSubtree(pRoot1->right,pRoot2);
    }
    bool dfs(TreeNode* root1,TreeNode* root2)
    {
        if(root1==NULL&&root2!=NULL)
            return false;
        if(root2==NULL)
            return true;
        if(root1->val!=root2->val)
            return false;
        return dfs(root1->left,root2->left)&&dfs(root1->right,root2->right);
    }
};
```

#### 23.二叉树的后续遍历

``` c++
class Solution {
public:
    bool VerifySquenceOfBST(vector<int> sequence) {
        if(sequence.size()<1)
            return false;
    return judge(sequence,0,sequence.size()-1);
    }
    bool judge(vector<int> seq,int pStart,int pEnd){
        if(pStart>=pEnd)
            return true;
        int index=0;
        int start=pStart;
        int end=pEnd-1;
        int root_val=seq[pEnd];
        while(seq[pStart]<root_val){
            pStart++;
        }
        int left=pStart-1;
        index=pStart;
        while(index<pEnd){
            if(seq[index]<root_val)
                return false;
            index++;
        }
        return judge(seq,start,left)&&judge(seq,left+1,end);
    }
};
```

            return;

        path.push_back(root->val);

            res.push_back(path);

        numPath(root->right,expectNum,num,path);
    }
};

#### 38.二叉树的深度

``` c++
//二叉树的先序遍历；
//每遍历一次当前root，那么就加一层，然后找出最大的层
class Solution {
public:
    int TreeDepth(TreeNode* pRoot)
    {
        if(pRoot==NULL)
            return 0;
        int res=0;
        deep(pRoot,0,res);
        return res;
    }
    void deep(TreeNode* root,int sum,int& max){
        if(root==NULL)
            return ;
        sum++;
        if(sum>max)
            max=sum;
        deep(root->left,sum,max);
        deep(root->right,sum,max);
        return ;
    }
};
```

#### 58.对称的二叉树

//对称就是不断的比较left和right
class Solution {
public:
    bool isSymmetrical(TreeNode* pRoot)
    {
    if(!pRoot)
        return true;
        return judge(pRoot->left,pRoot->right);
    }
    bool judge(TreeNode* root1,TreeNode* root2)
    {
        if(root1==NULL&&root2==NULL)
            return true;
        if((root1&&!root2)||(!root1&&root2))
            return false;
        if(root1->val!=root2->val)
            return false;
        return judge(root1->left,root2->right)&&judge(root1->right,root2->left);
    }

};

class Solution {

    {
        if(pRoot==NULL)
            return NULL;
        inorder(pRoot);
        if(k>res.size()||k<1)
            return NULL;
        return res[k-1];
    }
    void inorder(TreeNode* root)
    {
        if(root==NULL)
            return ;
        inorder(root->left);
        res.push_back(root);
        inorder(root->right);
        return ;
    }

    
};
