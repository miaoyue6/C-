## 5.栈实现队列

``` c++
class Solution
{
public:
    void push(int node) {
        stack1.push(node);
        
    }

    int pop() {
        while(!stack1.empty()){
            int tmp=stack1.top();
            stack2.push(tmp);
            stack1.pop();
        }
        int res;
        if(!stack2.empty())
        {
            res=stack2.top();
            stack2.pop();
        }
        while(!stack2.empty()){
            int tmp=stack2.top();
            stack1.push(tmp);
            stack2.pop();
        }
        return res;
            
        
    }

private:
    stack<int> stack1;
    stack<int> stack2;
};
```

#### 60.把二叉树打印成多行

``` c++
class Solution {
public:
        vector<vector<int> > Print(TreeNode* pRoot) {
            //vector<int> layer;
            vector<vector<int>> res;
            if(pRoot==NULL)
                return res;
            queue<TreeNode*> que;
            TreeNode* cur=pRoot;
            que.push(cur);
            int n=1;
            while(!que.empty()){
                int k=n;
                n=0;
                vector<int> layer;
                for(int i=0;i<k;i++)
                {
                    cur=que.front();
                    que.pop();
                    layer.push_back(cur->val);
                    if(cur->left!=NULL){
                        que.push(cur->left);
                        n++;
                    }
                    if(cur->right!=NULL){
                        que.push(cur->right);
                        n++;
                    }
                }
                res.push_back(layer);
            }
            return res;           
        }  
};
```
